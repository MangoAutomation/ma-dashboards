JsonStore
service in module maServices

Description

  • Provides service for reading and writing to the JsonStore within the Mango Database.
  • Used by <ma-json-store> directive.
  • All methods return $resource objects that can call the following methods availble to those objects:
  • $save
  • $remove
  • $delete
  • $get

Usage

 JsonStore.get({xid: newXid}).$promise.then(function(item) {
	return item;
 }, function() {
	var item = new JsonStore();
	item.xid = newXid;
	item.name = newXid;
	item.jsonData = $scope.value || {};
	angular.extend(item, $scope.item);
	return $q.when(item);
 }).then(function(item) {
	$scope.item = item;
 });
